Autofahrt praktikabel {
    Eigenes Auto vorhanden = ja OR Mitfahrgelegenheit möglich = ja [!],
} 

Mit eigenem Auto fahren sinnvoll {
    Eigenes Auto vorhanden = ja AND Anzahl Personen >= 3,
}   

Mitfahrgelegenheit möglich {
    Mitfahrgelegenheit = ja AND Anzahl Personen = 1,
    Menge an Gepäck = viel [--],
}


"Kann zum/vom Bahnhof laufen" {
    Menge an Gepäck = viel [--],
    "Entfernung zum/vom Bahnhof" = mittel AND Menge an Gepäck = mittel [--],
    "Entfernung zum/vom Bahnhof" = gering,
    Menge an Gepäck = wenig,
}

Zugfahrt praktikabel {
    Können Sie vom Zielbahnhof abgeholt werden? = ja OR "Kann zum/vom Bahnhof laufen" = true,
    Können Sie zum Abfahrtsbahnhof gebracht werden? = ja OR "Kann zum/vom Bahnhof laufen" = true,
}

Günstig allein Zug fahren {
    Anzahl Personen = 1 AND "Einzel-Bahnticketvergünstigung möglich" = ja,
}

Günstig als Gruppe Zug fahren {
    Anzahl Personen >= 3 AND "Gruppen-Bahnticketvergünstigung möglich" = ja,
}

Zugfahrt günstig {
    Günstig allein Zug fahren = High OR Günstig als Gruppe Zug fahren = trifft zu,
}

Autofahrt günstig {
    Mitfahrgelegenheit möglich = High OR Mit eigenem Auto fahren sinnvoll = ja,
}

Zugfahren {
    Zugfahrt praktikabel = ja,
    Zugfahrt günstig = ja,
}


Autofahren {
    Autofahrt praktikabel = ja,
    Autofahrt günstig = ja,
}
